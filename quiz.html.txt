function generateQuiz() {
  const quizContainer = document.getElementById('quiz');
  questions.forEach((q, index) => {
    const questionDiv = document.createElement('div');
    questionDiv.className = 'question';
    questionDiv.innerHTML = `<strong>प्रश्न ${index + 1}:</strong> ${q.q}`;
    const optionsDiv = document.createElement('div');
    optionsDiv.className = 'options';

    q.options.forEach((opt, i) => {
      const optionId = `q${index}_opt${i}`;
      const radio = `<input type="radio" name="question${index}" id="${optionId}" value="${String.fromCharCode(97 + i)}" />`;
      optionsDiv.innerHTML += `${radio} <label for="${optionId}">${opt}</label><br/>`;
    });

    questionDiv.appendChild(optionsDiv);
    quizContainer.appendChild(questionDiv);
  });
}

function submitQuiz() {
  let totalQuestions = questions.length;
  let score = 0;
  const wrongQuestions = [];

  questions.forEach((q, index) => {
    const selected = document.querySelector(`input[name="question${index}"]:checked`);
    if (selected && selected.value === q.answer) {
      score++;
    } else {
      const yourAnswerText = selected ? getAnswerText(selected.value, q.options) : "कोणतीही उत्तर नाही";
      wrongQuestions.push({
        question: q.q,
        yourAnswer: yourAnswerText,
        correctAnswer: getAnswerText(q.answer, q.options)
      });
    }
  });

  // निकाल तयार करणे
  let resultText = `एकूण गुण: ${score} / ${totalQuestions}\n\n`;

  if (wrongQuestions.length > 0) {
    resultText += "चुकीचे प्रश्न व त्यांची योग्य उत्तरे:\n\n";
    wrongQuestions.forEach((w, i) => {
      resultText += `प्रश्न ${i + 1}: ${w.question}\n`;
      resultText += `तुमचे उत्तर: ${w.yourAnswer}\n`;
      resultText += `योग्य उत्तर: ${w.correctAnswer}\n\n`;
    });
  } else {
    resultText += "सर्व उत्तरे योग्य आहेत!";
  }

  document.getElementById('result').innerText = resultText;
}

function getAnswerText(value, options) {
  const index = value.charCodeAt(0) - 97; // 'a' = 97
  return options[index];
}

// क्विझ तयार करणे
generateQuiz();
